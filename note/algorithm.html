# 冒泡排序

const bubbleSort = (arr) => {
  if(arr.length < 2) return arr
  for(let i=0; i<arr.length; i++){
    for(let j=0; j<arr.length-1-i; j++){
      if(arr[j+1] < arr[j]){   // 将相邻的较大的数移到后面
        [arr[j+1], arr[j]] = [arr[j], arr[j+1]]
      }
    }
  }
  return arr
}

# 选择排序

const selectionSort = (arr) => {
  if(arr.length < 2) return arr
  for(let i=0; i<arr.length; i++){
    let minIndex = i
    for(let j=i; j<arr.length; j++){
      if(arr[j] < arr[minIndex]){  // 寻找最小的数
        minIndex = j
      }
    }
    [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]]
  }
  return arr
}

# 插入排序

const insertionSort = (arr) => {
  if(arr.length < 2) return arr
  for(let i=0; i<arr.length-1; i++){
    let temp = arr[i+1]
    let preIndex = i;
    while(preIndex >= 0 && temp < arr[preIndex]){ // 将大于暂存值的数移到后面
      arr[preIndex+1] = arr[preIndex]
      preIndex--
    }
    arr[preIndex+1] = temp
  }
  return arr
}

# 希尔排序

const shellSort = (arr) => {
  let len = arr.length
  let gap = len >> 1
  while(gap > 0){
    for(let i=gap; i<len; i++){
      let temp = arr[i]
      let preIndex = i - gap  //以gap为间距进行插入排序，直到gap为0
      while(preIndex >= 0 && temp < arr[preIndex]){ 
        arr[preIndex+gap] = arr[preIndex]
        preIndex -= gap
      }
      arr[preIndex+gap] = temp
    }
    gap = gap >> 1
  }
  return arr
}

# 归并排序

递归方法

const merge = (left, right) => {
  let result = Array(left.length + right.length)
  for(let index=0, i=0, j=0; index<result.length; index++){
    if(i >= left.length) result[index] = right[j++]        //left放完了
    else if(j >= right.length) result[index] = left[i++]   //right放完了
    else if(left[i] > right[j]) result[index] = right[j++] 
    else result[index] = left[i++]    //将left[i]和right[j]中较小的放入
  }
  return result
}

const mergeSort = (arr) => {
  if(arr.length < 2) return arr
  let mid = arr.length >> 1
  const left = arr.slice(0, mid)
  const right = arr.slice(mid)
  return merge(mergeSort(left), mergeSort(right))
}

迭代方法

const mergeSort = (arr) => {
  if(arr.length < 2) return arr
  // 设置合并函数
  const merge = (ll, lr, rl, rr) => {
    if(rr >= arr.length) rr = arr.length - 1
    let i = ll, j = rl
    let temp = []
    while(i <= lr && j <= rr){
      temp.push(arr[i]<arr[j]?arr[i++]:arr[j++])
    }
    while(i <= lr) temp.push(arr[i++])
    while(j <= rr) temp.push(arr[j++])
    arr.splice(ll, temp.length, ...temp)
  }
  // step为数组内元素个数，step/2为左子区间元素个数
  for(let step = 2; step/2<arr.length; step*=2){
    // 每step个元素一组，组内前step/2和后step/2个元素进行合并
    for(let i=0; i<arr.length; i+=step){
      let mid = i + (step >> 1) - 1  // 左子区间元素个数为step/2
      if(mid + 1 < arr.length){   // 右子区间存在元素
        merge(i,mid,mid+1, i+step-1) // 右子区间最后一个元素下标为i-1+step
      }
    }
  }
  return arr
}

# 快速排序

const quickSort = (arr, start = 0, end = arr.length-1) => {
  if(start > end) return
  let i = start, j = end, mid = (start + end) >> 1
  if(arr[start] > arr[end]) [arr[start], arr[end]] = [arr[end], arr[start]]
  if(arr[mid] > arr[end]) [arr[mid], arr[end]] = [arr[end], arr[mid]]
  if(arr[mid] > arr[start]) [arr[mid], arr[start]] = [arr[mid], arr[start]]
  let base = arr[start]  //基准数取首、尾、中间三数中中间的数
  while(i < j){
    //右指针从右向左扫描，碰到第一个小于基准数的时候停住
    while (i < j && arr[j] >= base) j--
    //左指针从左向右扫描，碰到第一个大于基准数的时候停住
    while (i < j && arr[i] <= base) i++
    //交换左右指针所停位置的数
    [arr[i], arr[j]] = [arr[j], arr[i]]
  }  
  //最后交换基准数与指针相遇位置的数
  [arr[start],arr[i]] = [arr[i], arr[start]]
  quickSort(arr, start, i-1)
  quickSort(arr, i+1, end)
  return arr
}

# 堆排序

  • 大顶堆：arr[i] >= arr[2i+1] && arr[i] >= arr[2i+2]
  • 小顶堆：arr[i] <= arr[2i+1] && arr[i] <= arr[2i+2]
// 调整堆使其变成大顶堆
const adjustHeap = (arr, i, len) => {
  let temp = arr[i] // 先取出当前元素
  for(let j=2*i+1; j<len; j=2*j+1){ //从i结点的左子结点开始，也就是2i+1处开始
    if(j+1<len && arr[j]<arr[j+1]){ //如果左子结点小于右子结点，j指向右子结点
      j++
    }
    if(arr[j] > temp){//如果子节点大于父节点，将子节点值赋给父节点（不用进行交换）
      arr[i] = arr[j]
      i = j
    }else{
      break
    }
  }
  arr[i] = temp;//将temp值放到最终的位置
}

const heapSort = (arr) => {
  //1.构建大顶堆
  for(let i=arr.length-1>>1; i>=0; i--){
    adjustHeap(arr,i,arr.length)
  }
  //2.调整堆结构+交换堆顶元素与末尾元素
  for(let i=arr.length-1; i>0; i--){
    [arr[0], arr[i]] = [arr[i], arr[0]]
    adjustHeap(arr,0,i)
  }
  return arr
}

# 计数排序

const countingSort = (arr) => {
  const maxValue = Math.max(...arr)
  const bucket = new Array(maxValue + 1).fill(0)
  const res = []
  for(let i=0; i<arr.length; i++){
    bucket[arr[i]]++
  }
  for(let j=0; j<=maxValue; j++){
    while(bucket[j]>0){
      res.push(j)
      bucket[j]--
    }
  }
  return res
}
